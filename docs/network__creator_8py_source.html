254  def create_network(self, ip, mask_length, name=None):
255  """
256  Creates a virtual subnetwork
257 
258  Args:
259  ip: The given IP
260  mask_length: The network mask length of the subnetwork
261  name: The eventual name of the subnetwork
262 
263  """
264 
265  uid = len(self.subnetworks)
266 
267  # Name correspondency
268  if name:
269  result = self.is_name_existing('subnet', name)
270  if result:
271  raise NameAlreadyExists(name)
272  else:
273  name = f"<Untitled Network#ID:{uid}>"
274 
275  current = self.Network(ip, mask_length, uid, name)
276  current_netr = Utils.netr_to_literal(current.network_range)
277 
278  if self.ranges:
279  for sid in self.subnetworks:
280  subnet = self.subnetworks[sid]['instance']
281  subnetr = Utils.netr_to_literal(subnet.network_range)
282 
283  overlap = False
284  if current.mask_length == subnet.mask_length:
285  # Masks are equal
286  if current_netr['start'] == subnetr['start']:
287  overlap = True
288  else:
289  if current.mask_length < subnet.mask_length:
290  # New network mask is bigger, check if the existing subnetwork is inside
291  big = current_netr['start'].split('.')
292  small = subnetr['start']
293  else:
294  # New network is smaller that existing subnetwork, check if it is inside
295  small = current_netr['start']
296  big = subnetr['start'].split('.')
297 
298  if current.mask_length <= 8:
299  if int(big[0]) <= int(small.split('.')[0]):
300  overlap = True
301  elif 8 < current.mask_length <= 16:
302  if small.startswith(f"{big[0]}") and int(big[1]) <= int(small.split('.')[1]):
303  overlap = True
304  elif 16 < current.mask_length <= 24:
305  if small.startswith(f"{big[0]}.{big[1]}") and int(big[2]) <= int(small.split('.')[2]):
306  overlap = True
307  elif 24 < current.mask_length <= 32:
308  if int(big[3]) <= int(small.split('.')[3]):
309  overlap = True
310 
311  if overlap:
312  raise OverlappingError(current_netr, subnetr)
313 
314  self.subnetworks[uid] = {'instance': current, 'range': current.network_range}
315 
316  # adding to network ranges
317  self.ranges.append(current.network_range)
318  # also adding name if defined
319  if name:
320  self.subnets_names.append(name)
321 
322  return uid

360  def check_ip_availability(subnet_inst_, ip_):
361  """
362  Checks if an IP is available
363 
364  This function is a suicider: it will die if any of the tests fail. It will either raise
365  nettools.core.errors.IPOffNetworkRangeException or rth.core.errors.IPAlreadyAttributed
366 
367  Args:
368  subnet_inst_: The subnetwork instance
369  ip_: The IP that has to be checked
370  """
371 
372  # Checking that ip is effectively in range of the subnet
373  if isinstance(ip_, str):
374  ip_ = FourBytesLiteral().set_from_string_literal(ip_)
375 
376  mask = FourBytesLiteral().set_from_string_literal(Utils.mask_length_to_literal(subnet_inst_.mask_length))
377  inst = IPv4Network().init_from_fbl(ip_, mask)
378 
379  if inst.address_type != 1:
380  # means the address is either a network or a broadcast address
381  raise IPOffNetworkRangeException(str(ip))
382 
383  # then we check that ip is not used by any of the current routers
384  routers = subnet_inst_.routers
385  for r in routers:
386  if str(routers[r]) == str(ip_):
387  raise IPAlreadyAttributed(name, ip_, self.uid_to_name('router', r), str(router_name))
388 
389  router_uid = self.name_to_uid('router', router_name)
390 
391  for name in subnets_ips:
392  subnet_uid = self.name_to_uid('subnet', name)
393  subnet_inst = self.subnetworks[subnet_uid]['instance']
394  router_inst = self.routers[router_uid]
395 
396  subnet_ip = subnets_ips[name]
397 
398  # we want to attribute a "personalised" IP
399  if subnet_ip:
400  check_ip_availability(subnet_inst, subnet_ip)
401  ip = subnet_ip
402  # we will let the program set it for us
403  else:
404 
405  ip = subnet_inst.network_range['end']
406  while True:
407  ip = Utils.ip_before(ip)
408  try:
409  check_ip_availability(subnet_inst, ip)
410  break
411  except IPAlreadyAttributed:
412  continue
413 
414  subnet_inst.connect(router_uid, ip)
415  router_inst.connect(subnet_uid, ip)
416 
417  self.subnetworks[subnet_uid]['instance'] = subnet_inst
418  self.routers[router_uid] = router_inst
